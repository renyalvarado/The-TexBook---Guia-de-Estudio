\titulo{Ejercicio 19.11}

\enunciadoS ?`Qu\'e ocurre si se olvida escribir |&| en una ecuaci\'on
de |\eqalign|?

\bigskip

\respuesta{La l\'{\i}nea de la ecuaci\'on queda alineada a la
izquierda. Ejemplo:}

|$$\eqalign{a&=1\cr bb=2\cr ccc&=3\cr dddd&=4\cr}$$|

$$\eqalign{a&=1\cr bb=2\cr ccc&=3\cr dddd&=4\cr}$$

\bye
