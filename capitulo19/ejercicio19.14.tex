\titulo{Ejercicio 19.14}

\enunciadoS ?`Qu\'e ocurre si |\eqalignno| es sustituido por
|\eqalignno| en el \'ultimo ejemplo?

\bigskip

\respuestaS Se muestra el siguiente resultado:

$$\eqalign{
x&=y+z\cr
\noalign{\hbox{and}}
x^2&=y^2+z^2.\cr}$$

\noindent para

\

\noindent|$$\eqalign{|

\noindent|x&=y+z\cr|

\noindent|\noalign{\hbox{and}}|

\noindent|x^2&=y^2+z^2.\cr}$$|

\bye
