\titulo{Ejercicio 5.3}

\bigskip

\enunciado{Qu\'e piensa que ocurre si escribes lo siguiente:}
|\centerline{This information should be {centered}.}|

|\centerline So should this.|

\bigskip

\respuesta{La primera instrucci\'on centrar\'a el texto, mientras que
la segunda s\'olo centrar\'a la letra 'S' de la palabra 'So' y
colocar\'a el resto del texto en la siguiente l\'{\i}nea (el texto no
estar\'a centrado).}

\medskip

|\centerline{This information should be {centered}.}|

\centerline{This information should be {centered}.}

\smallskip

|\centerline So should this.|

\centerline So should this.


\bye
