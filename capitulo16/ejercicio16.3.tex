\titulo{Ejercicio 16.3}

\bigskip

\enunciadoS ?`Qu\'e diferencia existe (si la hay), entre 
`|$x + _2F_3$|' y `|$x + {}_2F_3$|'?

\bigskip

\respuesta{La diferencia es que en la primera ecuaci\'on el $2$ es
sub\'{\i}ndice del operador $+$ y en la segunda el n\'umero es
sub\'{\i}ndice izquierdo de la $F$\/.}

\bye
