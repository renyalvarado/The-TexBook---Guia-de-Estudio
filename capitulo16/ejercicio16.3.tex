\titulo{Ejercicio 16.3}

\bigskip

\enunciado{?`Qu\'e diferencia existe (si la hay), entre `{\tt \$x + \_2F\_3\$}' y `{\tt \$x + \char`\{\char`\}\_2F\_3\$}'?}

\bigskip

\respuesta{La diferencia es que en la primera ecuaci\'on el $2$ es sub\'{\i}ndice del operador $+$ y en la segunda el n\'umero es sub\'{\i}ndice izquierdo de la $F$\/.}
