\titulo{Ejercicio 16.4}

\enunciado{Describa las diferencias entre las salidas de `{\tt \$\char`\{x\^{}y\char`\}\^{}z\$}' y `{\tt \$x\^{}\char`\{y\^{}z\char`\}\$}'.}

\bigskip

\respuesta{Primera expresi\'on: ${x^y}^z$; segunda expresi\'on: $x^{y^z}$. En la primera, la $x$ y la $z$ tienen el mismo tama\~no de la fuente. En la segunda, la $z$ es mas peque\~na.}
