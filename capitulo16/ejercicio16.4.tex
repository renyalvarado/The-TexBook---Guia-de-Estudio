\titulo{Ejercicio 16.4}

\bigskip

\enunciadoS Describa las diferencias entre las salidas de
`|${x^y}^z$|' y `|$x^{y^z}$|'.

\bigskip

\respuesta{Primera expresi\'on: ${x^y}^z$; segunda expresi\'on:
$x^{y^z}$. En la primera, la $x$ y la $z$ tienen el mismo tama\~no de
la fuente. En la segunda, la $z$ es mas peque\~na.}

\bye
