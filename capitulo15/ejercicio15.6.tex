\titulo{Ejercicio 15.6}

\bigskip

\enunciadoS Pruebe sus conocimientos de los registros de {\TeX}
indicando los resultados de cada uno de los siguientes comandos cuando
son ejecutados en secuencia:

|\count1=50| \kern0.4cm 
|\dimen2=\count1pt| \kern0.4cm |\divide\count1 by 8|

|\skip2=-10pt plus\count1fil minus\dimen2|

|\multiply\skip2 by-\count1| \kern0.4cm 
|\divide\skip2 by \dimen2 \count6 \skip2|

|\skip1=.5\dimen2 plus\skip2 minus\count\count1fill|

|\multiply\skip2 by\skip1| \kern0.4cm 
|\advance\skip1 by-\skip2|

\bigskip

\respuesta{Los resultados de los c\'alculos son: }

|\count1=50| \kern0.4cm 
\count1=50 \the\count1

|\dimen2=\count1pt| \kern0.4cm
\dimen2=\count1pt \the\dimen2

|\divide\count1 by 8| \kern0.4cm
\divide\count1 by 8 \the\count1

\medskip

|\skip2=-10pt plus \count1fil minus \dimen2| \kern0.4cm 
\skip2=-10pt plus \count1fil minus \dimen2 \the\skip2

\medskip

|\multiply\skip2 by -\count1| \kern0.4cm 
\multiply\skip2 by -\count1 \the\skip2

|\divide\skip2 by \dimen2| \kern0.4cm 
\divide\skip2 by \dimen2 \the\skip2

|\count6=\skip2| \kern0.4cm 
\count6=\skip2 \the\count6

\medskip

|\skip1=.5\dimen2 plus\skip2  minus \count\count1fill| \kern0.4cm 
\skip1=.5\dimen2 plus\skip2  minus \count\count1fill  \the\skip1

\medskip

|\multiply\skip2 by \skip1| \kern0.4cm 
\multiply\skip2 by \skip1 \the\skip2

|\advance\skip1 by \skip2| \kern0.4cm 
\advance\skip1 by \skip2 \the\skip1

\bye
