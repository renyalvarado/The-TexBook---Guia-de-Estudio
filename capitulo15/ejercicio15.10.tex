\titulo{Ejercicio 15.10}

\bigskip

\enunciadoS ?`Cu\'al es el valor del registro |\box5| despu\'es de los
siguientes comandos: ?

|\setbox5=\hbox{A}| \kern 0.4cm 
|\setbox5=\hbox{\copy\unbox5\box5\unhcopy5}|

\bigskip

\respuestaS El valor de |\box5| es |\hbox{\hbox{A}A}|, ya que luego de
aplicar |\unhbox5|, la caja del registro $5$ queda vac\'{\i}a, por lo
que el resto de los comandos copian nulo al final de la caja de texto.

\bye
