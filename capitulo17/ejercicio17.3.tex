\titulo{Ejercicio 17.3}

\bigskip

\enunciado{?`Qu\'e sorpresa obtuvo B.~L.~User cuando escribi\'o `{\tt \$\$x = (y\^\/2\comando{\over} k+1)\$\$}'?}

\bigskip

\respuesta{Obtuvo el siguiente resultado: }

$$ x = (y^2\over k+1)  $$

\noindent ya que el comando \comando{\over} considera todo lo que est\'a a su izquierda como parte del numerador.
