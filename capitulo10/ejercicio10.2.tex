\titulo{Ejercicio 10.2}

\bigskip

\enunciado{Ordene esas seis t\'{\i}picas dimensiones en orden, de la m\'as peque
~na a la m\'as grande.}

\noindent Dimensiones: |3 in|; |-.013837 in|; |0.mm|; |29 pc|; |+ 42,1
dd|; |123456789sp|

\bigskip

\respuesta{}

\medskip
|-.013837in|

|0.mm = 0in|

|+ 42,1dd = 45.05pt = 0.62in|

|3in|

|29pc = 348pt = 4.82in|

|123456789sp = 1883pt = 26.06in|

\bye
