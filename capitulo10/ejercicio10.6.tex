|\magnificaction+\magstep1|

|\font\first=cmr10 scaled\magstep1|

|\font\second=cmr10 at 12truept|

\bigskip

\respuesta{}
|\first| ser\'a la fuente |cmr10| con 14.4pt 

|\second| ser\'a la fuente |cmr10| con 12pt
