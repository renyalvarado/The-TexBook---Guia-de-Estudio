\titulo{EJERCICIO 14.27}

\enunciado{?`Cu\'al es la penalizaci\'on entre las l\'{\i}neas de un p\'arrafo
de dos l\'{\i}neas?}

\respuesta{La penalizaci\'on se calcula en funci\'on de \comando{\interlinepenalty}
mas \comando{\clubpenalty} mas \comando{\widowpenalty} y si la primera l\'{\i}nea 
tiene gui\'on al final agregamos \comando{\brokenpenalty}.}
