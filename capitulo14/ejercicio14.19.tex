\titulo{EJERCICIO 14.19}
\enunciado{Suponga que uno de los casos enumerados contin\'ua por dos o m\'as
p\'arrafos.?`C\'omo usar\'{\i}a \comando{\item} para mantener la identaci\'on
de los siguientes p\'arrafos?}

\medskip

\respuesta{Para cada uno de los p\'arrafos enumerados despu\'es del primero,
se deber\'{\i}a agregar el comando \comando{\item\char`\{\char`\}}. Por ejemplo:
}
\item{1.- } Am\'erica.
\item{} Venezuela.
\item{} Colombia.


\item{2.- } Europa.
\item{} Espa\~na.
\item{} Francia.
\end
